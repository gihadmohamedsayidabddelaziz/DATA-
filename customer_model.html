# Function to calculate the Interquartile Range (IQR) and identify outliers
def remove_outliers(customer, cols):
    for col in cols:
        q1 = customer[col].quantile(0.25)
        q3 = customer[col].quantile(0.75)
        iqr = q3 - q1
        upper_limit = q3 + 1.5 * iqr
        lower_limit = q1 - 1.5 * iqr
        
        outliers = customer[(customer[col] < lower_limit) | (customer[col] > upper_limit)]
      
        customer = customer[~customer.index.isin(outliers.index)]
        
        # Print information about removed outliers
        print(f"Outliers removed from {col}: {outliers.shape[0]}")
    
    return customer
